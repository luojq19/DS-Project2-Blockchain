\subsection*{To open the project}
Run the following code in powershell:
\begin{center}
	\code{go run *.go <nodeID> <workload>}
\end{center}
where \code{<nodeID>} is a string (default \code{3000}, \code{3001}, \code{3002},...) to identify stored blockchains, and \code{<workload>} is a file (default \code{test.data}, \code{test1.data}, \code{test2.data},...) with series of operations contained:
\begin{center}
	\begin{tabular}{|l|l|}
		\hline
		codes&operations\\
		\hline
		\hline
		\code{**createblockchain <name>} & create a blockchain with wallet \code{<name>}\\
		\hline
		\code{**createwallet <name>} & create a wallet (i.e. an address) named \code{<name>}\\
		\hline
		\code{**getbalance <name>} & get the balance of the wallet named \code{<name>}\\
		\hline
		\code{**listaddresses} & list all addresses from the wallet file\\
		\hline
		\code{**printchain} & print all the blocks of the blockchain\\
		\hline
		\code{**reindexUTXO} & rebuild the UTXO set\\
		\hline
		\code{**send <name1> <name2> <amount>} & a transaction from \code{<name1>} to \code{<name2>} of \code{<amount>}\\
		\hline
		\code{**startnode <mode> <name>} & \begin{tabular}{l}
			\code{<mode>}=\code{normal}:only listen for txs\\
			\code{<mode>}=\code{thread}:both listen for txs and do operations\\
			\code{<mode>}=\code{miner}:only mine, with wallet \code{<name>}\\
		\end{tabular}\\
		\hline
		\code{**sleep <period>} & sleep for \code{<period>} (waiting for some operations)\\
		\hline
	\end{tabular}
\end{center}
